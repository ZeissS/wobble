  3 /**
  4  * <div id="contacts" class="widget">
  5  *    <div class="whoami">
  6  *    </div>
  7  *    <div class="actions">
  8  *      <button id="contacts_add">Add</button>
  9  *      <button id="user_change_name">Change my name</button>
 10  *    </div>
 11  *    <ul class="contactslist">
 12  *    </ul>
 13  *  </div>
 14  */
 15 function JQueryContactsView() {
 16   var template =
 17     '<div class="whoami"></div>' +
 18     '<div class="actions">' +
 19     ' <button id="contacts_add">Add</button>' +
 20     ' <button id="show_my_profile">Show my profile</button>' +
 21     '</div>' +
 22     '<ul class="contactslist"></ul>';
 23   this.e = $('<div></div>').addClass('widget').attr('id', 'contacts').appendTo('#widgets');
 24 
 25   // NOTE: Makes sure that thie contactsList does not start with a 100% width
 26   // and thus prvents the other widgets from beeing rendered
 27   this.e.css('width', '180px');
 28 
 29   this.e.append(template);
 30     this.$whoami = $(".whoami", this.e);
 31     this.$actions = $(".actions", this.e);
 32     this.$contactsList = $(".contactslist", this.e);
 33 
 34   // UI Event Handlers
 35   this.$whoami.click($.proxy(function() {
 36      this.fireWhoamiClicked();
 37   }, this));
 38   $("#contacts_add").click($.proxy(function() {
 39     var contactEmail = window.prompt("Enter the new contact's email address");
 40     if (contactEmail !== null) {
 41       this.onAddContact(contactEmail);
 42     }
 43   }, this));
 44   $("#show_my_profile").click($.proxy(function() {
 45      var that = this;
 46      that.fireWhoamiClicked();
 47   }, this));
 48 
 49   // On a window.resize event wait for the transformations to finish (should be done in 300ms) and recalc height
 50   function on_window_resize() {
 51     var t = this;
 52     window.setTimeout(function() {
 53       t.onResize();
 54     }, 350);
 55   }
 56   BUS.on('window.resize', on_window_resize, this);
 57   on_window_resize.call(this); // Fire it once initially (with a delay)
 58 
 59 };
 60 JQueryContactsView.prototype = new ContactsDisplay();
 61 JQueryContactsView.prototype.constructor = JQueryContactsView;
 62 
 63 // Methods 
 64 JQueryContactsView.prototype.onResize = function() {
 65   var viewHeight = this.e.innerHeight();
 66   var offsetX = this.$whoami.outerHeight() + this.$actions.outerHeight()
 67 
 68   this.$contactsList.css('height', viewHeight - offsetX);
 69 };
 70 
 71 JQueryContactsView.prototype.fireWhoamiClicked = function() {
 72   var that = this;
 73   BUS.fire('contact.clicked', {
 74       'contact': API.user(),
 75       'actions': [
 76         {title: 'Change my name', callback: function() {
 77             var newName = window.prompt("What should your new name be?");
 78         if (newName !== null) {
 79           that.onNameChange(newName);
 80         }
 81         }},
 82         {title: 'Change password', callback: function() {
 83            var p1 = window.prompt('What should your new password be?');
 84            if (p1 !== null) {
 85              var p2 = window.prompt('And once again to be sure:');
 86              if (p2 !== null) {
 87                if (p1 === p2) {
 88                    that.onPasswordChange(p1);
 89                } else {
 90                    window.alert('Your two passwords do not match. Try again.');   
 91                }
 92              }   
 93            }
 94         }}
 95       ]
 96    });
 97 };
 98 JQueryContactsView.prototype.renderContacts = function (list) {
 99   this.$contactsList.empty();
100 
101   jQuery.each(list, $.proxy(function(i, contact) {
102     var template = "<li class=contact title='{{email}}'>" +
103             "<div class='usericon usericon{{size}}'>" +
104             "<div><img src='http://gravatar.com/avatar/{{{img}}}?s={{size}}' width={{size}} height={{size}}></div>" +
105             "<div class='status {{online}}'></div>" +
106             "</div>" +
107             "<span class=name>{{name}}</span>" +
108             "</li>";
109     $(Mustache.to_html(template, {
110         size: 20,
111         email: contact.email,
112         name: contact.name,
113         img: contact.img,
114         online: contact.online == 1 ? 'online' : 'offline'
115     })).appendTo(this.$contactsList).click($.proxy(function() {
116       this.onContactClick(contact);
117     }, this));
118   }, this));
119 };
120 JQueryContactsView.prototype.renderWhoAmI = function(user) {
121   this.$whoami.empty();
122   var template = "<img title='That is you!' src='http://gravatar.com/avatar/{{{img}}}?s=32' width=32 height=32> <span class=name>{{name}}</span>";
123   this.$whoami.append(Mustache.to_html(template, user));
124 };
