1 "use strict";
  2 function AbstractLoginView() {};
  3 AbstractLoginView.prototype = new EventBUS();
  4 
  5 AbstractLoginView.prototype.show = function() {
  6   this.e.css('display', '');
  7 };
  8 AbstractLoginView.prototype.hide = function() {
  9   this.e.css('display', 'none');
 10 };
 11 AbstractLoginView.prototype.destroy = function() {
 12   this.e.empty();
 13   this.e.remove();
 14 };
 15 AbstractLoginView.prototype.setEnabled = function(enabled) {
 16   var buttons = $("button,input", this.e);
 17   if (enabled) {
 18     buttons.removeAttr('disabled');
 19   } else {
 20     buttons.attr('disabled', 'disabled');
 21   }
 22 };
 23 
 24 // Register Event Handlers
 25 AbstractLoginView.prototype.fireLogin = function(email, password) {
 26   this.fire('loginview.login', {'email': email, 'password': password});
 27 };
 28 AbstractLoginView.prototype.fireRegister = function(email, password) {
 29   this.fire('loginview.register', {'email': email, 'password': password});
 30 };
 31 AbstractLoginView.prototype.onLogin = function(callback) {
 32   this.on('loginview.login', callback);
 33 };
 34 AbstractLoginView.prototype.onRegister = function(callback) {
 35   this.on('loginview.register', callback);
 36 };
