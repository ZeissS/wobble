1 "use strict";
  2 function ContactsDetailDisplay() {};
  3 ContactsDetailDisplay.prototype.show = function(contact) {};
  4 ContactsDetailDisplay.prototype.addAction = function(label, callback) {};
  5 ContactsDetailDisplay.prototype.onClose = function() {};
  6 
  7 
  8 
  9 function ContactsDetailPresenter(display, model, eventName) {
 10   this.display = display;
 11   this.model = model;
 12 
 13   BUS.on(eventName, function(data) {
 14     var theUser = data.user || data.contact;
 15     display.show(theUser);
 16 
 17     // Add action buttons from provided params
 18     if (data.actions) {
 19       for (var x in data.actions) {
 20         display.addAction(
 21           data.actions[x].title,
 22           data.actions[x].callback
 23         );
 24       }
 25     }
 26     // Also add an 'Add contact' button, if user is not in model and not ourself
 27     if (theUser.id !== API.user_id() && model.get(theUser.id) === undefined) {
 28       display.addAction('Add as contact', function() {
 29         BUS.fire('contacts.adduser', theUser.email);
 30         display.hide();
 31       });
 32     }
 33 
 34   });
 35 }
