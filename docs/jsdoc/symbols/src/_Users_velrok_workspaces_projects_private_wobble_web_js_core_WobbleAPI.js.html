  4 // API Wrapper functions
  5 var WobbleAPI = function(rpc, callback) {
  6   if (!rpc) {
  7     throw new Error('RPC object is required for WobbleAPI object');
  8   }
  9   this.rpc = rpc;
 10   this._user = undefined;
 11   this.apikey = {
 12     'value': localStorage.getItem('WOBBLEAPI_APIKEY'),
 13     'get': function() {
 14       return this.value;
 15     },
 16     'set': function(apikey) {
 17       this.value = apikey;
 18       if (!apikey) {
 19         localStorage.removeItem('WOBBLEAPI_APIKEY');
 20       } else {
 21         localStorage.setItem('WOBBLEAPI_APIKEY', apikey);
 22       }
 23     }
 24   };
 25 
 26   if (this.apikey.get()) {
 27     this.refreshUser(callback);
 28   } else {
 29     if(callback) callback(null);
 30   }
 31 };
 32 WobbleAPI.prototype.destroy = function() {};
 33 /**
 34  * Adds the apikey to the parameters and call the doRPC method of the rpc object.
 35  *
 36  * @see this.rpc
 37  */
 38 WobbleAPI.prototype.doRPC = function(name, params, callback) {
 39   if(this.apikey.get()) {
 40     if(typeof(params) == "function") {
 41       callback = params;
 42       params = {
 43         'apikey': this.apikey.get()
 44       };
 45     } else if(params == undefined) {
 46       params = {
 47         'apikey': this.apikey.get()
 48       };
 49     } else {
 50       params.apikey = this.apikey.get();
 51     }
 52   }
 53   this.rpc.doRPC(name, params, callback);
 54 };
 55 
 56 WobbleAPI.prototype.refreshUser = function(callback) {
 57   this.user_get($.proxy(function(err, user) {
 58     if (!err) {
 59       this._user = user;
 60       BUS.fire('api.user', user);
 61       if (callback) {
 62         return callback(user);
 63       }
 64     }
 65   }, this));
 66 };

 69 /** Builds an ID by combining the user_id with the current time. */
 70 WobbleAPI.prototype.generate_id = function() {
 71   var id = this.user_id() + "-" + (new Date().getTime()) + "-" + Math.ceil(Math.random() * 100000);
 72   return id;
 73 };
 74 WobbleAPI.prototype.user_id = function() {
 75   return this._user ? this._user.id : null;
 76 };

 78   return this._user;

 95 /* REGISTER / LOGIN ----------------- */
 96 WobbleAPI.prototype.register = function(email, password, callback) {
 97   var that = this;
 98   this.rpc.doRPC('user_register', {'email': email, 'password': password}, function(err, result) {
 99     if(!err && result.apikey) {
100       that.apikey.set(result.apikey);
101     }
102     if(callback)
103       return callback(err, result);
104   });
105 };
106 WobbleAPI.prototype.login = function(email, password, callback) {
107   var that = this;
108   this.doRPC('user_login', {'email': email, 'password': password}, function(err, result) {
109     if(!err && result.apikey)
110       that.apikey.set(result.apikey);
111 
112     if(callback)
113       return callback(err, result);
114   });
115 };
116 WobbleAPI.prototype.signout = function(callback) {
117   var that = this;
118   this.doRPC('user_signout', function(err, result) {
119     that.apikey.set(undefined);
120     if(callback)
121       return callback(err, result);
122   });
123 };
