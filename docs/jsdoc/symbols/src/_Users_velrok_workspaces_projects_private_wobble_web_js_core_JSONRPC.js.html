1 "use strict";
  2 
  3 // RPC Wrapper (JSON-RPC 2.0 - http://json-this.RPC.org/wiki/specification)
  4 var JSONRPC = function(url) {
  5   this.url = url;
  6   this.idSequence = 1;
  7   this.stateWaiting = [];
  8 }
  9 /**
 10  * Mark this RPC object as aborted, so any result that comes in is not propagated further.
 11  * This means no callbacks will be called anymore.
 12  */
 13 JSONRPC.prototype.destroy = function(name) {
 14   this.aborted = true;
 15 };
 16 
 17 /**
 18  * Notifications are calls, where no response is expected/wished.
 19  */
 20 JSONRPC.prototype.doNotification = function(name, args) {
 21   RPC._call(null, name, args, null);
 22 };
 23 /**
 24  * Normal RPC call.
 25  */
 26 JSONRPC.prototype.doRPC = function(name, args, callback) {
 27   if (arguments.length == 2) {
 28     callback = args;
 29     args = null;
 30   }
 31   var requestId = this.idSequence;
 32   this.idSequence++;
 33 
 34   this._call(requestId, name, args, callback);
 35 };
 36 JSONRPC.prototype._call = function(requestId, name, args, callback) {
 37   var that = this;
 38 
 39   var body = {
 40     jsonrpc: "2.0",
 41     method: name,
 42   };
 43   if (args) {
 44     body.params = args;
 45   }
 46   if (requestId) {
 47     body.id = requestId;
 48   }
 49 
 50   var ajaxSettings = {
 51     type:'POST',
 52     cache: false,
 53     data: JSON.stringify(body),
 54     dataType: "json",
 55     processData: false,
 56     contentType: 'application/json; charset=utf-8',
 57     success: function(data, textStatus, jqXHR) {
 58       if(that.aborted)
 59         return;
 60       if (!callback)
 61         return;
 62 
 63       if (data == undefined) {
 64         var error = {'message': 'Empty response'};
 65         var errorHandled = callback(error);
 66         if (!errorHandled) {
 67           BUS.fire('rpc.error', {
 68             request: body,
 69             error: error
 70           });
 71         }
 72       }
 73       else if (data.error) {
 74         var errorHandled = callback(data.error);
 75         if (!errorHandled) {
 76           BUS.fire('rpc.error', {
 77             request: body,
 78             error: data.error
 79           });
 80         }
 81       } else {
 82         callback(undefined, data.result);
 83       }
 84     }
 85   };
 86   if (callback) {
 87     ajaxSettings.error = function(jqXHR, textStatus, errorThrown) {
 88       if(that.aborted)
 89         return;
 90 
 91       BUS.fire('rpc.connectionerror', {text: textStatus, error: errorThrown});
 92       callback(errorThrown);
 93     };
 94   }
 95 
 96   var that = this;
 97   var req = $.ajax(this.url, ajaxSettings).always(function() {
 98     that.stateWaiting = jQuery.grep(that.stateWaiting, function(areq, i) {
 99       return areq !== req;
100     });
101     BUS.fire('rpc.queue.length', that.stateWaiting.length);
102   });
103   this.stateWaiting.push(req);
104   BUS.fire('rpc.queue.length', this.stateWaiting.length);
105 };
