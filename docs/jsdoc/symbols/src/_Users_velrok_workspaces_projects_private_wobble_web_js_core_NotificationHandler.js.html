1 "use strict";
  2 
  3 /*
  4  * Queries the API / Server in an endless loop and fires BUS events for each returned notification
  5  *
  6  * Fires:
  7  * o api.notifications
  8  *
  9  * Listens:
 10  * o api.user
 11  */
 12 function NotificationHandler() {
 13   // Start fetching, when user is available
 14   if (window.API && window.API.user() != null) {
 15     this.fetch_notifications();
 16   } else {
 17     BUS.on('api.user', function() {
 18       this.fetch_notifications();
 19     }, this);
 20   }
 21 };
 22 
 23 NotificationHandler.prototype.fetch_notifications = function (next_timestamp) {
 24   var that = this;
 25 
 26   API.get_notifications(next_timestamp, function(err, notifications) {
 27     if (notifications && notifications.messages) {
 28       for (var i = 0; i < notifications.messages.length; i++) {
 29         BUS.fire('api.notification', notifications.messages[i]);
 30       }
 31     }
 32     that.timeoutId = window.setTimeout(function() {
 33       that.fetch_notifications(notifications ? notifications.next_timestamp : null);
 34     }, 500);  // Check in 0.5secs
 35   });
 36 }
 37 
 38 NotificationHandler.prototype.destroy = function() {
 39   if (this.timeoutId) {
 40     window.clearTimeout(this.timeoutId);
 41   }
 42 };
