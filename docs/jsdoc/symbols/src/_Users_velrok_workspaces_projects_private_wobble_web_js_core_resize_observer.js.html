1 /**
  2  * Fires a 'window.resize' BUS event, when the window gets resized.
  3  */
  4 (function($) {
  5   "use strict";
  6   var $window = $(window);
  7   function get_size() {
  8     return {w: $window.width(), h: $window.height()};
  9   };
 10 
 11   $(document).ready(function() {
 12     // Fire initial event to setup UI
 13     var old_size = get_size();
 14 
 15     window.setTimeout(function() {
 16       BUS.fire('window.resize', {before: old_size, 'to': old_size});
 17     }, 10); // Add a little delay so the UI can be constructed
 18 
 19     // Listen for changes
 20     $window.resize(function() {
 21       var new_size = get_size();
 22       BUS.fire('window.resize', {
 23         'before': old_size,
 24         'to':  new_size
 25       });
 26       old_size = new_size;
 27     });
 28   });
 29 })(jQuery);
