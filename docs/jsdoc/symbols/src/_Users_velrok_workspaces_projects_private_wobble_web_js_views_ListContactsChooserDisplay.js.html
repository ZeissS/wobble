1 "use strict";
  2 
  3 /**
  4  * A contactschooser dialog similar to the original Google Wave version.
  5  * The dialog shows all contacts who are not yet added to the topic in a list.
  6  * The user can filter the list with a textfield at the top. 
  7  * The left/right keys navigate inside the filter.
  8  * The up/down keys navigate in the contacts list.
  9  * By clicking a contact or pressing [Enter] when selected, the callback onAddContact() gets fired.
 10  */
 11 function ListContactsChooserDisplay(relativeTo) {
 12   this.e = $('<div></div>').attr('id', 'contactschooser').appendTo('body'); // The element to use
 13   this.e.css('display', 'none');
 14   this.e.addClass('dialog');
 15 
 16   this.relativeTo = relativeTo;
 17 
 18   this.contacts = [];
 19   this.selectedContact = null;
 20 };
 21 ListContactsChooserDisplay.prototype = new ContactsChooserDisplay;
 22 ListContactsChooserDisplay.prototype.constructor = ListContactsChooserDisplay;
 23 ListContactsChooserDisplay.prototype.show = function(title, contacts) {
 24   this.title = title;
 25   this.contacts = contacts;
 26   this.render();
 27 };
 28 ListContactsChooserDisplay.prototype.render = function() {
 29   var $filterText, $contactList;
 30 
 31   // Generate the content
 32   var template =  '<div id="contactschooser_title">{{title}}</div>' +
 33           '<div class="buttons"><button class=button_close>x</button></div>' +
 34           '<div id="contactschooser_filter"><input type=text id="contactschooser_filter_text"></div>' +
 35           '<ul id="contactschooser_list">'
 36           '</ul>';
 37   this.e.empty().append(Mustache.to_html(template, {
 38     'title': this.title
 39   }));
 40   $contactList = $('#contactschooser_list', this.e);
 41   if (this.contacts.length == 0) {
 42     $contactList.append('<li>No contacts</li>');
 43   } else {
 44     jQuery.each(this.contacts, $.proxy(function(i, contact) {
 45       var template = "<li class=contact title='{{email}}'>" +
 46               "<div class='usericon usericon{{size}}'>" +
 47               "<div><img src='http://gravatar.com/avatar/{{{img}}}?s={{size}}' width={{size}} height={{size}}></div>" +
 48               "<div class='status {{online}}'></div>" +
 49               "</div>" +
 50               "<span class=name>{{name}}</span>" +
 51               "</li>";
 52       var $li = $(Mustache.to_html(template, {
 53           size: 20,
 54           email: contact.email,
 55           name: contact.name,
 56           img: contact.img,
 57           online: contact.online == 1 ? 'online' : 'offline'
 58       })).attr('id', 'contactchooser-contact-' + contact.id).appendTo($contactList);
 59       $li.data('contact', contact);
 60 
 61       $li.click($.proxy(function() {
 62         // Move cursor to clicked and then one down
 63         this.setSelectedContact(contact);
 64         this.navigateNextContact();
 65 
 66         // Contact was clicked
 67         this.onAddContact(contact);
 68         $li.detach(); // Remove row after adding it
 69         $filterText.focus(); // Refocus the text input, where we capture the keyboard events
 70       }, this));
 71 
 72       if (i == 0) {
 73         // Autoselect the first element
 74         this.setSelectedContact(contact);
 75       }
 76     }, this));
 77   }
 78 
 79   // Install button-listeners
 80   $('.button_close', this.e).click($.proxy(function() {
 81     this.close();
 82   }, this));
 83 
 84   $filterText = $("#contactschooser_filter_text");
 85   $filterText.keydown($.proxy(function(e) {
 86     if (e.which == 27) {
 87       // Close dialog on escape
 88       this.close();
 89     }
 90     else if (e.which == 38) {
 91       // Naviagte up
 92       this.navigatePreviousContact();
 93       e.preventDefault();
 94     }
 95     else if (e.which == 40) {
 96       // Navigate down
 97       this.navigateNextContact();
 98       e.preventDefault();
 99     }
100     else if (e.which == 13) {
101       e.preventDefault();
102       $('#contactchooser-contact-' + this.selectedContact.id).click(); // Simulate clicking on it
103     }
104 
105   }, this));
106   $filterText.keyup($.proxy(function(e) {
107     // Refresh filtered list
108     this.refreshFilteredContactList($filterText.val());
109   }, this));
110 
111   // Position it relative to this.relativeTo
112   if (this.relativeTo) {
113     var relativeElem = $(this.relativeTo);
114     var pos = relativeElem.offset();
115     this.e.css('top', pos.top).css('left', pos.left - (this.e.width() * 0.75));
116   }
117 
118   // Finally, show it
119   this.e.css('display', '');
120 
121   // And focus the textfield
122   $filterText.focus();
123 };
124 ListContactsChooserDisplay.prototype.close = function() {
125   this.e.empty().css('display', 'none'); // Clear and hide
126   this.onClose(); // Notify caller
127 };
128 ListContactsChooserDisplay.prototype.setSelectedContact = function(contact) {
129   $('.active', this.e).removeClass('active');
130   if (contact != null) {
131     $('#contactchooser-contact-' + contact.id, this.e).addClass('active');
132   }
133 
134   this.selectedContact = contact;
135 };
136 ListContactsChooserDisplay.prototype.navigateNextContact = function() {
137   var element = $("#contactchooser-contact-" + this.selectedContact.id);
138 
139   // Find the next element which is visible
140   do {
141     element = element.next();
142   } while (element.css('display') == 'none');
143 
144   if (element) {
145     this.setSelectedContact(element.data('contact'));
146   }
147 };
148 ListContactsChooserDisplay.prototype.navigatePreviousContact = function() {
149   var element = $("#contactchooser-contact-" + this.selectedContact.id);
150 
151   // Find the next element which is visible
152   do {
153     element = element.prev();
154   } while (element.css('display') == 'none');
155 
156   if (element) {
157     this.setSelectedContact(element.data('contact'));
158   }
159 };
160 ListContactsChooserDisplay.prototype.refreshFilteredContactList = function(filterText) {
161   var firstContact = null;
162 
163   $(".contact", this.e).css('display', ''); // Show all contacts
164 
165   filterText = filterText.toLowerCase();
166 
167   for (var i = 0; i < this.contacts.length; i++) {
168     var contact = this.contacts[i];
169     // Show all elements
170     if (!(contact.name.toLowerCase().indexOf(filterText) >= 0 || contact.email.indexOf(filterText) >= 0)) {
171       // text not found in name or email
172       var $contact = $("#contactchooser-contact-"+ contact.id);
173       $contact.css('display', 'none');
174 
175       // If the selectedContact is now hidden, set selectedContact to null
176       if (contact == this.selectedContact) {
177         this.setSelectedContact(null);
178       }
179     } else {
180       if (firstContact == null) {
181         firstContact = contact;
182       }
183     }
184   };
185 
186   if (this.selectedContact == null && firstContact != null) {
187     // No contact selected, mark the first visible one
188     this.setSelectedContact(firstContact);
189   }
190 };
