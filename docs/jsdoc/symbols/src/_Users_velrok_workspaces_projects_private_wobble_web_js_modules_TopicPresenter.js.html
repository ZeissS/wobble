 27 function TopicModel() {
 28   var that = this;
 29   var topic = null;
 30 
 31   // Getters & Setters -------------------------------
 32   that.setTopic = function(new_topic) {
 33     topic = new_topic;
 34   };
 35   that.getTopic = function() {
 36     return topic;
 37   };
 38 
 39   // Methods ------------------------------------------
 40   that.removePost = function(post) {
 41     topic.posts = _.filter(topic.posts, function(post2) {
 42       return post.id === post2.id;
 43     });
 44   };
 45   that.addPost = function(post) {
 46     topic.posts.push(post);
 47   };
 48 
 49 };
 50 TopicModel.prototype.createReply = function(post) {
 51   return {
 52     id: API.generate_id(),
 53     parent: post.id,
 54     locked: false,
 55     content: '\n<br>\n<br>',
 56     revision_no: 1,
 57     users: [API.user_id()],
 58     deleted: 0,
 59     unread: 0
 60   };
 61 };
 62 TopicModel.prototype.addUserToPost = function(post, user) {
 63   var found = false;
 64   for (var i = 0; i < post.users.length; i++) {
 65     if (post.users[i] === user.id) {
 66       found = true;
 67     }
 68   }
 69   if (!found) {
 70     post.users.push(user.id);
 71     // We can assume here, that the user is part of topic.users, otherwise he shouldn't see this post
 72   }
 73 };
 74 TopicModel.prototype.addUser = function(user) {
 75   if (_.contains(this.getTopic().readers, user)) {
 76     this.getTopic().readers.push(user);
 77   }
 78 };
 79 TopicModel.prototype.removeUser = function(user, callback) {
 80   var topic = this.getTopic();
 81 
 82   API.topic_remove_user(topic.id, user.id, function(err, result) {
 83     if (!err) {
 84       topic.readers = _.filter(topic.readers, function(tuser) {
 85         return user.id != tuser.id; // Filter the given user
 86       });
 87     }
 88     callback(err, result);
 89   });
 90 };
 91 TopicModel.prototype.getUserIds = function() {
 92   var topic = this.getTopic();
 93   var result = [];
 94 
 95   for (var i = 0; i < topic.readers.length; i++) {
 96     result.push(topic.readers[i].id);
 97   }
 98 
 99   return result;
100 };

105 function TopicPresenter(view, model) {
106   this.view = view;
107   this.model = model;
108 
109   view.setEnabled(false);
110 
111   //// ---- View Event Callbacks ------------------------------------------------------
112   var that = this;
113   view.onInviteUserAction = function() {
114     BUS.fire('contacts.chooser.open', {
115       'multiple': true,
116       'remove_contacts': model.getUserIds(),
117       'on_add': function (contact) {
118         API.topic_add_user(model.getTopic().id, contact.id, function(err, data) {
119           model.addUser(contact);
120           view.renderTopic(model.getTopic());
121         });
122       },
123       'on_close': function() {
124         BUS.fire('topic.changed', model.getTopic().id);
125       }
126     });
127   };
128   view.onUserClicked = function(user) {
129     var actions = [];
130  
131     if (user.id != API.user_id()) {
132       actions.push({title: 'Remove from Topic', callback: function() {
133         that.model.removeUser(user, function(err, result) {
134           view.renderTopic(model.getTopic());
135         });
136       }});
137 
138     }
139 
140     BUS.fire('topic.user.clicked', {
141       'user': user,
142       'actions': actions
143     });
144   };
145   view.onPostClicked = function(post) {
146     // remove unread class on click + mark read on server side
147     if (post.unread == 1) {
148       var topic = model.getTopic();
149 
150       post.unread = 0;
151       post.locked = true;
152       view.renderPost(model.getTopic(), post);
153 
154       API.post_change_read(model.getTopic().id, post.id, 1, function() {
155         post.locked = false;
156 
157         // Still our topic? => Refresh View
158         if (model.getTopic().id == topic.id) {
159           view.renderPost(topic, post);
160         }
161         BUS.fire('topic.post.changed', model.getTopic().id);
162       });
163     }
164   };
165   view.onStartPostEdit = function(post) {
166     // Create the lock
167     API.post_change_lock(model.getTopic().id, post.id, 1, function(err, success) {
168       if (err || !success) {
169         alert('Failed to create lock for post. Try again later or refresh your browser.');
170       } else {
171         model.addUserToPost(post, API.user());
172         view.renderPost(model.getTopic(), post);
173       }
174     });
175   };
176   view.onStopPostEdit = function(post, content) {
177     post.locked = true; // Lock post until saved
178     post.content = content;
179     view.renderPost(model.getTopic(), post);
180     API.post_edit(model.getTopic().id, post.id, content, post.revision_no, function(err, result) {
181       if (!err) {
182         post.revision_no = result.revision_no;
183 
184         BUS.fire('topic.post.changed', model.getTopic().id);
185       }
186       post.locked = false;
187       view.renderPost(model.getTopic(), post);
188     });
189   };
190 
191   view.onReplyPost = function(post) {
192     var newPost = model.createReply(post);
193     newPost.locked = true;
194     API.post_create(model.getTopic().id, newPost.id, newPost.parent, 0, function(err, data) {
195       newPost.locked = false;
196       view.renderPost(model.getTopic(), newPost);
197     });
198     model.addPost(newPost);
199     view.renderTopic(model.getTopic());
200     view.openEditor(newPost);
201   };
202   view.onIntendedReplyPost = function(post) {
203     var newPost = model.createReply(post);
204     newPost.intended_post = 1;
205     newPost.locked = true;
206     API.post_create(model.getTopic().id, newPost.id, newPost.parent, 1, function(err, data) {
207       newPost.locked = false;
208       view.renderPost(model.getTopic(), newPost);
209     });
210     model.addPost(newPost);
211     view.renderTopic(model.getTopic());
212     view.openEditor(newPost);
213   };
214 
215   view.onDeletePost = function(post) {
216     post.locked = true;
217     API.post_delete(model.getTopic().id, post.id, function(err, result) {
218       post.locked = false;
219       that.refreshTopic();
220     });
221     model.removePost(post);
222     view.clear();
223     view.renderTopic(model.getTopic());
224   };
225 
226   view.onMoveToArchive = function() {
227     API.topic_set_archived(model.getTopic().id, 1, function(err, result) {
228     });
229     view.clear();
230     model.setTopic(null);
231   };
232   view.onMoveToInbox = function() {
233     API.topic_set_archived(model.getTopic().id, 0, function(err, result) {
234     });
235     view.clear();
236     model.setTopic(null);
237   };
238 
239 
240   /**
241    * BUS Handlers
242    */
243 
244   // Fired by TopicsPresenter
245   BUS.on('topic.selected', function(topicId) {
246     if (model.getTopic() != null && model.getTopic().id == topicId) {
247       return;
248     }
249     model.setTopic({id: topicId});
250     view.setLoadingState();
251     that.refreshTopic();
252   });
253   // Fired when a new topic got created by TopicsPresenter
254   BUS.on('topic.topic.created', function(topicId) {
255     model.setTopic({id: topicId});
256     view.setLoadingState();
257 
258     that.refreshTopic(function() { // Refreshs the data based on the set ID
259       view.openEditor(that.model.getTopic().posts[0]);
260     });
261   });
262 
263   BUS.on('api.notification', function(data) {
264     // Somebody else changed our topic
265     if (model.getTopic() != null && (
266       data.type == 'topic_changed' && data.topic_id == model.getTopic().id ||
267       data.type == 'post_deleted' && data.topic_id == model.getTopic().id ||
268       data.type == 'post_changed' && data.topic_id == model.getTopic().id))
269     {
270       that.refreshTopic();
271     }
272   });
273 };
274 TopicPresenter.prototype.refreshTopic = function(callback) {
275   if (this.model.getTopic() == null)
276     return;
277 
278   var that  = this;
279 
280   API.load_topic_details(this.model.getTopic().id, function(err, topicDetails) {
281     if (err) { // Just clear the view, when the data could not be loaded.
282       console.log('Failed to load topic. Clearing view.');
283       that.view.clear();
284       return true;
285     }
286     var modelTopic = that.model.getTopic();
287     if (modelTopic !== null && topicDetails !== undefined && topicDetails.id == modelTopic.id) { // Check that we still want to see this data
288       that.setSelectedTopic(topicDetails);
289       if (callback) {
290         callback();
291       }
292     }
293   });
294 };
295 /**
296  * Change the underlying topic
297  */
298 TopicPresenter.prototype.setSelectedTopic = function(topicDetails) {
299   if (topicDetails === this.model.getTopic()) {
300     return;
301   }
302   this.model.setTopic(topicDetails);
303   this.view.renderTopic(topicDetails);
304 };
