  3 function ContactsModel() {
  4   this.cache = {};
  5   this.user = API.user();
  6 
  7   var that = this;
  8   BUS.on('api.user', function(user) {
  9     that.user = user;
 10   });
 11 };
 12 ContactsModel.prototype.findByEmail = function (email) {
 13   var result = undefined;
 14   for (var i = 0; i < this.cache.length; i++) {
 15     var user = this.cache[i];
 16     if (user.email === email) {
 17       result = user;
 18     }
 19   }
 20   return result;
 21 };
 22 ContactsModel.prototype.getContacts = function (callback) {
 23   var that = this;
 24   API.get_contacts(function(err, data) {
 25     if (data) {
 26       for (var i = 0; i < data.length; i++) {
 27         that.put(data[i]);
 28       }
 29     }
 30     callback(err, data);
 31   });
 32 };
 33 ContactsModel.prototype.put = function (contact) {
 34   this.cache[contact.id] = contact;
 35 };
 36 ContactsModel.prototype.get = function (id) {
 37   return this.cache[id];
 38 }
 39 ContactsModel.prototype.addNewContact = function(contactEmail, callback) {
 40   API.add_contact(contactEmail, callback);
 41 };
 42 ContactsModel.prototype.getUser = function() {
 43   return this.user;
 44 }
 45 ContactsModel.prototype.removeUserFromRooster = function(userId, callback) {
 46   var that = this;
 47   API.contact_remove(userId, function(err, data) {
 48     // Update the model
 49     delete that.cache[userId];
 50 
 51     // Call the callback
 52     callback(err, data);
 53   });
 54 }

 57 /**
 58  * The prototype for the ContactsDisplay used by the ContactsPresenter.
 59  */
 60 function ContactsDisplay() {}
 61 // Event Callbacks (Will be set by presenter)
 62 ContactsDisplay.prototype.onAddContact = function(contactEmail) {};
 63 ContactsDisplay.prototype.onContactClick = function(contact) {};
 64 ContactsDisplay.prototype.onNameChange = function(new_name) {};
 65 ContactsDisplay.prototype.onPasswordChange = function(new_password) {};
 66 // Methods
 67 ContactsDisplay.prototype.renderContacts = function (list) {};
 68 ContactsDisplay.prototype.renderWhoAmI = function(user) {};
 69 ContactsDisplay.prototype.showMessage = function(message) { window.alert(message); };
 70 
 71 
 72 function ContactsPresenter(display, model) {
 73   this.display = display;
 74   this.model = model;
 75 
 76   this.refreshContacts(); // Load initially
 77   if (model.getUser()) {
 78     this.display.renderWhoAmI(model.getUser());
 79   }
 80 
 81   var that = this;
 82 
 83   // Timers & Data Loading  ----------------------------------
 84   var timer = setTimeout(function() {
 85     that.refreshContacts();
 86   }, 60 * 1000);  // Reload contact list once a minute
 87 
 88   // Button Handlers  ---------------------------------------------------
 89   display.onAddContact = function(contactEmail) {
 90     that.addUserByEmail(contactEmail);
 91   };
 92   display.onContactClick = function(contact) {
 93     BUS.fire('contact.clicked', {
 94       'contact': contact,
 95       'actions': [
 96         {title: 'Remove Contact', callback: function() {
 97           that.removeUserFromRooster(contact.id);
 98         }}
 99       ]
100     });
101   };
102   display.onNameChange = function(newName) {
103     API.user_change_name(newName, function(err, result) {
104       API.refreshUser();
105     });
106   };
107   display.onPasswordChange = function(newPassword) {
108      API.user_change_password(newPassword, function(err, result) {
109          if (result) {
110              window.alert('Password changed successfully.');
111          }
112      });
113   };
114 
115   // BUS Handler  ---------------------------------------------------
116   BUS.on('contacts.refresh', function() {
117     this.refreshContacts();
118   }, this);
119   BUS.on('api.user', function(user) {
120     this.display.renderWhoAmI(user);
121   }, this);
122   BUS.on('api.notification', function(message) {
123     if (message.type == 'user_signout' || message.type == 'user_online') { // We receive this message only, because we are on his contacts list
124       this.refreshContacts();
125     }
126   }, this);
127   BUS.on('contacts.adduser', function(userEmail) {
128     this.addUserByEmail(userEmail);
129   }, this);
130 
131   return that;
132 }
133 // Methods ---------------------------------------------------
134 ContactsPresenter.prototype.removeUserFromRooster = function(userId) {
135   var that = this;
136   this.model.removeUserFromRooster(userId, function(err) {
137     that.refreshContacts();
138   });
139 }
140 ContactsPresenter.prototype.addUserByEmail = function(email) {
141   var that = this;
142   this.model.addNewContact(email, function(err, data) {
143     if (data) {
144       that.refreshContacts();
145       that.display.showMessage('Contact added!');
146     } else {
147       that.display.showMessage('Contact could not be added.');
148     }
149   });
150 };
151 ContactsPresenter.prototype.refreshContacts = function () {
152   var display = this.display;
153   this.model.getContacts(function(err, data) {
154     if (!err) {
155       display.renderContacts(data);
156     }
157   });
158 };
