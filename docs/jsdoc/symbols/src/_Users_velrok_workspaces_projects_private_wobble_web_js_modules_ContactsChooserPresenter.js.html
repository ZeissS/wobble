1 "use strict";
  2 /**
  3  * The display for the ContactsChooserPresenter
  4  */
  5 function ContactsChooserDisplay() {}
  6 // Methods
  7 ContactsChooserDisplay.prototype.show = function(title, contacts) {};
  8 ContactsChooserDisplay.prototype.close = function() {};
  9 ContactsChooserDisplay.prototype.setPresenter = function(presenter) {
 10   this.presenter = presenter;
 11 }
 12 // Callbacks
 13 ContactsChooserDisplay.prototype.onAddContact = function(contact) {};
 14 ContactsChooserDisplay.prototype.onClose = function(contact) {};
 15 
 16 /**
 17  * The ContactsChooserPresenter. Listens for BUS events to show it's UI. Use the ContactsModel.
 18  * 
 19  * @param display A ContactsChooserDisplay Implementation
 20  * @param model A ContactsModel
 21  */
 22 function ContactsChooserPresenter(display, model) {
 23   this.display = display;
 24   this.model = model;
 25 
 26   this.display.setPresenter(this);
 27 
 28   BUS.on('contacts.chooser.open', function(config) {
 29     var options = _.defaults(config, {
 30       'multiple': false,
 31       'on_add': function(contact) {},
 32       'on_close': function() {},
 33       'remove_contacts': []
 34     });
 35     this.open(options);
 36   }, this);
 37 
 38   // Autoclose the display when the topic changes
 39   BUS.on('topic.selected', function() {
 40     this.display.close();
 41   }, this);
 42 };
 43 /**
 44  * Starts a loop where the user can add users. For each added user the options.on_add callback is executed.
 45  * 
 46  * options: {
 47  *  on_add: function(contact) {},
 48  *  on_close: function() {},
 49  *  multiple: true | false,
 50  *  remove_contacts: [user_id]
 51  * }
 52  */
 53 ContactsChooserPresenter.prototype.open = function(options) {
 54   var display = this.display;
 55   this.model.getContacts(function(err, contacts) {
 56     if (!err) {
 57       var showableContacts = _.filter(contacts, function(contact) {
 58         var shouldBeRemoved = _.contains(options.remove_contacts, contact.id);
 59         return !shouldBeRemoved;
 60       });
 61 
 62       display.onAddContact = function(contact) {
 63         options.on_add(contact);
 64         if (!options.multiple) display.close();
 65       };
 66       display.onClose = function() {
 67         options.on_close();
 68       };
 69       display.show('Add participant', showableContacts);
 70     }
 71   });
 72 };
