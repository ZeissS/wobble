1 (function() {
  2   var isDev = localStorage && localStorage.getItem('WOBBLE_DEV');
  3   var files = [];
  4   var WobbleAppName = (function() {
  5     if (localStorage) {
  6       // An enforced app has priority over the UA detection
  7       var appName = localStorage.getItem('WOBBLE_ENFORCE_APP');
  8       if (appName == "mobile") {
  9         return 'WobbleMobileClient';
 10       } else if(appName == "desktop") {
 11         return 'WobbleDesktopClient';
 12       }
 13     }
 14 
 15     // If the UA detects a mobile browser, use that
 16     var ua = navigator.userAgent;
 17     if (ua.match(/android/i) ||
 18       ua.match(/iphone/i)) {
 19       return 'WobbleMobileClient';
 20     }
 21 
 22     // Fallback: Standard desktop version
 23     return 'WobbleDesktopClient';
 24   })();
 25 
 26 
 27   // External Libraries
 28   if (isDev) {
 29     files.push(
 30       'js/ext/underscore.js',
 31       'js/ext/jquery-1.7.1.js',
 32       'js/ext/mustache.js'
 33     );
 34   } else {
 35     files.push(
 36       'js/ext/underscore-min.js',
 37       'js/ext/jquery.min.js',
 38       'js/ext/mustache.min.js'
 39     );
 40   }
 41 
 42   // Wobble Core
 43   files.push(
 44     'js/core/localcache.js',
 45     'js/core/resize_observer.js',
 46     'js/core/EventBUS.js',
 47     'js/core/JSONRPC.js',
 48     'js/core/WobbleAPI.js',
 49     'js/core/NotificationHandler.js',
 50     'js/core/BasicClient.js'
 51   )
 52   // Wobble Modules
 53   files.push(
 54     'js/modules/ContactsPresenter.js',
 55     'js/modules/ContactsChooserPresenter.js',
 56     'js/modules/ContactsDetailPresenter.js',
 57     'js/modules/TopicPresenter.js',
 58     'js/modules/TopicListPresenter.js',
 59 
 60     // Views
 61     'js/views/jQueryTopicListView.js',
 62     'js/views/jQueryTopicView.js',
 63     'js/views/JQueryContactsView.js',
 64     'js/views/jQueryContactsDetailDisplay.js',
 65     'js/views/ListContactsChooserDisplay.js',
 66 //    'js/views/SimpleContactsChooserDisplay.js',
 67 
 68     'js/login/LoginModel.js',
 69     'js/login/AbstractLoginView.js',
 70     'js/login/DesktopLoginView.js',
 71     'js/login/MobileLoginView.js',
 72     'js/login/LoginPresenter.js'
 73   );
 74 
 75   // Load classes based selected App
 76   if (WobbleAppName.match(/mobile/i)) {
 77     files.push(
 78       'js/mobile/MobileNavigator.js',
 79       'js/mobile/WobbleMobileClient.js'
 80     );
 81   } else {
 82     files.push(
 83       'js/desktop/DesktopClientHeader.js',
 84       'js/desktop/WobbleDesktopClient.js'
 85     );
 86   }
 87 
 88   for(var i = 0; i < files.length; i++) {
 89     document.write('<script src="' + files[i] + '"></script>');
 90   }
 91 
 92   // Wait until all scripts are loaded
 93   window.onload = function() {
 94     window.WobbleApp = new window[WobbleAppName];
 95     window.WobbleApp.bootstrap();
 96   };
 97 })();
