  3 /**
  4  * A simple ContactsChooserDisplay Implementation using only the standard 
  5  * javascript window.prompt() function. 
  6  */
  7 function SimpleContactsChooserDisplay() {}
  8 SimpleContactsChooserDisplay.prototype = new ContactsChooserDisplay;
  9 SimpleContactsChooserDisplay.prototype.constructor = SimpleContactsChooserDisplay;
 10 SimpleContactsChooserDisplay.prototype.show = function(title, contacts) {
 11   var that = this;
 12   window.setTimeout(function() { // Do this async
 13     var contactNameOrEmail = window.prompt(title);
 14     if (contactNameOrEmail != null) {
 15       var matchingContacts = jQuery.grep(contacts, function(contact, index) {
 16         return contact.name == contactNameOrEmail || contact.email == contactNameOrEmail;
 17       });
 18       if (matchingContacts.length > 0) {
 19         that.onAddContact(matchingContacts[0]);
 20         return;
 21       }
 22     }
 23     // No contact entered or found, closing
 24     that.close();
 25   });
 26 };
 27 SimpleContactsChooserDisplay.prototype.close = function() {
 28   this.onClose();
 29 };
