 38 jQueryTopicsView.prototype.renderTopicList = function renderTopicList(topics, prepend) {
 39   // Update document title
 40   var user = API.user();
 41   if (user && topics) {
 42     var unreadTopics = jQuery.grep(topics, function(topic) {
 43       return topic.post_count_unread > 0;
 44     }).length;
 45     if (unreadTopics == 0) {
 46       document.title = user.name + " - Wobble";
 47     } else {
 48       document.title = "(" + unreadTopics + ") " + user.name + " - Wobble";
 49     }
 50   }
 51 
 52   // Render to html list
 53   if (topics.length == 0) {
 54     this.renderText('No topics here. Try to create one :)');
 55   }
 56   for (var i = 0; i < topics.length; ++i) {
 57     this.renderTopic(topics[i], prepend);
 58   }
 59 };
 60 jQueryTopicsView.prototype.renderTopic = function renderTopic(topic, prepend) {
 61   var template = '<li id="{{id}}" class="topic_header">' +
 62            ' <div class="abstract"></div>' +
 63            (topic.post_count_unread == 0 ?
 64               ' <div class="messages">{{total}} msgs</div>' :
 65               ' <div class="messages"><div class=unread>{{unread}}</div> of {{total}}</div>')
 66             +
 67            ' <div class="time">{{time}}</div>' +
 68            ' <div class="users">{{#users}}<img title="{{name}}" src="http://gravatar.com/avatar/{{img}}?s=32" width="32" height="32">{{/users}}</div>' +
 69            '</li>';
 70   var that = this;
 71   var $li = $(Mustache.to_html(template, {
 72     'id': 'topic-' + topic.id,
 73     'users': topic.users.slice(0,3) /* Make sure we only have 3 users */,
 74     'unread': topic.post_count_unread,
 75     'total': topic.post_count_total,
 76     'time': this.renderTopicTimestamp(topic.max_last_touch)
 77   })).data('topic', topic);
 78 
 79   $li.on('click', function() {
 80     var topic = $(this).data('topic');
 81     if (topic) {
 82       that.onTopicClicked(topic);
 83     }
 84   });
 85 
 86   var abstract = $(".abstract", $li).html(topic.abstract);
 87 
 88   if (topic.post_count_unread > 0) {
 89     abstract.css('font-weight', 'bold');
 90   }
 91 
 92   if (prepend) {
 93     $li.prependTo(this.$topics);
 94   } else {
 95     $li.appendTo(this.$topics)
 96   }
 97 };
 98 jQueryTopicsView.prototype.renderTopicTimestamp = function renderTopicTimestamp(timestamp) {
 99   if (!timestamp) {
100     return "";
101   }
102   // NOTE: This format the date in the german way (localized): dd.mm.yyyy hh24:mi
103   var createdAt = new Date(timestamp * 1000), now = new Date();
104   var hours = createdAt.getHours();
105   if (hours < 10) {
106     hours = "0" + hours;
107   }
108   var minutes = createdAt.getMinutes();
109   if (minutes < 10) {
110     minutes = "0" + minutes;
111   }
112   var time = hours + ":" + minutes;
113 
114   var month = createdAt.getMonth() + 1;
115   if (month < 0){
116     month = "0" + month;
117   }
118 
119   if (createdAt.getYear() == now.getYear()) {
120     if (createdAt.getMonth() == now.getMonth() &&
121       createdAt.getDate() == now.getDate()) { // This post is from today, only show the time
122       return time;
123     } else {
124       // this post is at least from this year, show day + month
125       return createdAt.getDate() + "." + month + ".";
126     }
127   } else {
128     return createdAt.getDate() + "." + month + "."+ (1900 + createdAt.getYear());
129   }
130 };
