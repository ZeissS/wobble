 11 WobbleDesktopClient.prototype = new BasicClient;
 12 WobbleDesktopClient.prototype.init = function(user) {
 13   if (user != null) {
 14     this.initApp();
 15   } else {
 16     this.initLogin();
 17   }
 18 };
 19 WobbleDesktopClient.prototype.initLogin = function() {
 20   var that = this;
 21 
 22   this.loginView = new DesktopLoginView();
 23   this.loginModel = new LoginModel();
 24 
 25   this.loginPresenter = new LoginPresenter(
 26     this.loginView,
 27     this.loginModel,
 28     function() {
 29       that.loginView.destroy(); // Remove the LoginView from the document
 30 
 31       API.refreshUser();
 32 
 33       // More code to show the actuall app
 34       that.initApp();
 35     }
 36   );
 37 };

 39 WobbleDesktopClient.prototype.initApp = function() {
 40   // Models
 41   this.contactsModel = new ContactsModel();
 42 
 43   // Views
 44   this.topHeader = new DesktopClientHeader();
 45 
 46   // Presenter
 47   this.contactsChooserPresenter = new ContactsChooserPresenter(
 48       new ListContactsChooserDisplay('#topic_invite_user'),
 49       this.contactsModel
 50   );
 51   this.contactsPresenter = new ContactsPresenter(new JQueryContactsView(), this.contactsModel);
 52   this.topicsPresenter = new TopicListPresenter(new jQueryTopicsView(), window.localcache.getCache());
 53 
 54   this.contactsDetailPresenter = new ContactsDetailPresenter(new jQueryContactsDetailDisplay(100, 100), this.contactsModel, 'contact.clicked');
 55   this.topicUserDetailPresenter = new ContactsDetailPresenter(new jQueryContactsDetailDisplay(600, 100), this.contactsModel, 'topic.user.clicked');
 56 
 57   this.topicPresenter = new TopicPresenter(new jQueryTopicView(), new TopicModel());
 58 
 59   // Ok, all done. Lay it out
 60   this.doLayout();
 61 
 62   // Recalculate the position of the widgets, when window is resized
 63   BUS.on('window.resize', function(data) {
 64     this.doLayout(data);
 65   }, this);
 66 };
 67 WobbleDesktopClient.prototype.onRPCError = function(err) {
 68   var doReload = window.confirm('Whoooops! Something went wrong! We will reload now, ok?');
 69   if (doReload)
 70     window.location.reload();
 71 };

 73 /**
 74  * Calculate the widths of the three main widgets.
 75  */
 76 WobbleDesktopClient.prototype.doLayout = function(data) {
 77   if (!data) {
 78     data = {'to': {w: $(window).width(),h:$(window).height() }};
 79   }
 80   // TODO: Move this into some kind of Layout object?
 81   function minWidth(e) {
 82     var sW = e.css('min-width');
 83     return parseFloat(sW);
 84   }
 85   var headline = $("#headline");
 86   var topics = $("#topics_wrapper");
 87   var topic = $("#topic_wrapper");
 88   var contacts = $("#contacts");
 89 
 90   contacts.css('left', 5);
 91 
 92   var rightPart = data.to.w - (parseFloat(contacts.css('left')) + contacts.width());
 93 
 94   var topicsLeft = contacts.width() + 15;
 95   var topicsWidth = Math.max(rightPart * 0.4, minWidth(topics));
 96   topics.css('left', topicsLeft);
 97   topics.width(topicsWidth);
 98 
 99   var topicLeft = topicsLeft + topicsWidth + 10;
100   var topicWidth = Math.max(data.to.w - topicLeft - 15, minWidth(topics));
101   topic.css('left', topicLeft);
102   topic.width(topicWidth);
103 
104   var borderBottom = 35;
105   contacts.height(data.to.h - headline.height() - borderBottom);
106   topics.height(data.to.h - headline.height() - borderBottom);
107   topic.height(data.to.h - headline.height() - borderBottom);
108 };
