 24 WobbleMobileClient.prototype.init = function(user) {
 25   if (user != null) {
 26     this.initApp();
 27   } else {
 28     this.initLogin();
 29   }
 30 };
 31 WobbleMobileClient.prototype.initLogin = function() {
 32   var that = this;
 33 
 34   this.loginView = new MobileLoginView();
 35   this.loginModel = new LoginModel();
 36 
 37   this.loginPresenter = new LoginPresenter(
 38     this.loginView,
 39     this.loginModel,
 40     function() {
 41       that.loginView.destroy(); // Remove the LoginView from the document
 42 
 43       API.refreshUser();
 44 
 45       // More code to show the actuall app
 46       that.initApp();
 47     }
 48   );
 49 };

 51 WobbleMobileClient.prototype.initApp = function() {
 52   this.$widgets = $("#widgets");
 53 
 54   // DataModel
 55   this.contactsModel = new ContactsModel();
 56   this.topicModel = new TopicModel();
 57 
 58   // Create the navigator
 59   this.navigatorView = new MobileNavigation(this);
 60 
 61   // Create the views
 62   this.contactsView = new JQueryContactsView();
 63   this.topicListView = new jQueryTopicsView();
 64   this.topicView = new jQueryTopicView();
 65 
 66   // Create the presenters
 67   this.contactsPresenter = new ContactsPresenter(this.contactsView, this.contactsModel);
 68   this.contactsDetailPresenter = new ContactsDetailPresenter(new jQueryContactsDetailDisplay(20, 20), this.contactsModel, 'contact.clicked');
 69   this.topicUserDetailPresenter = new ContactsDetailPresenter(new jQueryContactsDetailDisplay(20, 20), this.contactsModel, 'topic.user.clicked');
 70   this.topicsPresenter = new TopicListPresenter(this.topicListView, window.localcache.getCache());
 71   this.topicPresenter = new TopicPresenter(this.topicView, this.topicModel);
 72   this.contactsChooserPresenter = new ContactsChooserPresenter(
 73       new ListContactsChooserDisplay(),
 74       this.contactsModel
 75   );
 76 
 77   // Now show the contacts list
 78   this.onNavigation('navOverview');
 79 
 80   this.doLayout();
 81 
 82   BUS.on('topic.topic.created', function(topicId) {
 83     this.onNavigation('navTopic');
 84   }, this);
 85 };
 86 WobbleMobileClient.prototype.doLayout = function() {
 87   var map = {'position':'absolute', 'width':'', 'left': '0px', 'right':'0px'};
 88   this.contactsView.e.css(map);
 89   this.topicListView.e.css(map);
 90   this.topicView.e.css(map);
 91 };

 93 WobbleMobileClient.prototype.onNavigation = function(targetId) {
 94   $(">*", this.$widgets).detach(); // Detach() does not destroy the event handlers
 95 
 96   window.location.hash = "";
 97   if (targetId == 'navContacts') {
 98     // Show the ContactsList
 99     this.contactsView.e.appendTo(this.$widgets);
100 
101   } else if (targetId == 'navOverview') {
102     // Show the TopicsList
103     this.topicListView.e.appendTo(this.$widgets);
104   } else if (targetId == 'navTopic') {
105     // Show the Topic itself
106     this.topicView.e.appendTo(this.$widgets);
107   } else if (targetId == 'navLogout') {
108     API.signout(function(err, result) {
109       window.location.reload();
110     });
111   } else {
112     // Unknown error
113     alert('Unknown navigation id: ' + targetId);
114   }
115 };
