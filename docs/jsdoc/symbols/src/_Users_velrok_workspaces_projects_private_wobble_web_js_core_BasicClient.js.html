  3 function BasicClient() {};
  4 
  5 /**
  6  * Called when all dependencies are loaded. Initializes the basic objects 
  7  * like API, RPC.
  8  */
  9 BasicClient.prototype.bootstrap = function() {
 10   $('<div></div>').attr('id', 'widgets').appendTo($('body'));
 11 
 12   // Show a reload dialog, when an RPC error occurs
 13   var rpcOperations = 0;
 14   BUS.on('rpc.queue.length', function(length) {
 15     rpcOperations = length;
 16   });
 17 
 18   BUS.on('rpc.error', function(err) {
 19     this.onRPCError(err);
 20   }, this);
 21   BUS.on('rpc.connectionerror', function(err) {
 22     this.onRPCError(err);
 23   }, this);
 24 
 25   var that = this;
 26   $(window).bind('beforeunload', function() {
 27     // We ignore 1 operation, because there is always a notification fetcher working
 28     if (rpcOperations > 1) {
 29       return 'There are operations pending. Are you sure you want to close now?'
 30     }
 31     that.unload();
 32   });
 33 
 34   // Initialize the API and app, when user data is loaded
 35   window.RPC = new JSONRPC('./api/endpoint.php');
 36   window.API = new WobbleAPI(
 37     window.RPC,
 38     function(user) {
 39       that.preinit(user);
 40     }
 41   );
 42 }
 43 BasicClient.prototype.preinit = function(user) {
 44   this.notificationFetcher = new NotificationHandler();
 45 
 46   $('body').empty().append('<div id=widgets></div>');
 47 
 48 
 49   var appStateSelectedTopicId = window.location.hash ? window.location.hash.substring(1) : null;
 50   // Fire a select event, when the hash changes, e.g. by the user clicking on something or
 51   // Using the back button ...
 52   $(window).on('hashchange', function() {
 53     var s = window.location.hash;
 54     if (s.length > 0) {
 55       s = s.substring(1);
 56       if (s !== appStateSelectedTopicId) {
 57         appStateSelectedTopicId = s;
 58         BUS.fire('topic.selected', appStateSelectedTopicId);
 59       }
 60     }
 61   });
 62 
 63   BUS.on('topic.selected', function(topicId) {
 64     if (window.location.hash !== "#" + topicId) {
 65       appStateSelectedTopicId = topicId;
 66       window.location.hash = topicId; // Note the current topicId in the URL, so its visible for the user and we can work with it on page reloads
 67     }
 68   });
 69 
 70   this.init(user);
 71 
 72   // Ok, init is done. Now fire the initial topic.select, if given ;)
 73   if (appStateSelectedTopicId !== null) {
 74     BUS.fire('topic.selected', appStateSelectedTopicId);
 75   }
 76 };
 77 BasicClient.prototype.init = function(user) {
 78   // Overwrite in Implementation
 79 };
 80 BasicClient.prototype.unload = function(user) {
 81   console.log('Unload detected. Marking backend object as destroyed.');
 82   // Overwrite in Implementation
 83   if (this.notificationFetcher) this.notificationFetcher.destroy();
 84   if (window.RPC) window.RPC.destroy();
 85   if (window.API) window.API.destroy();
 86 };
 87 
 88 BasicClient.prototype.onRPCError = function(err) {};
