  3 /**
  4  * The loginview should support the following features:
  5  * - Login by email + password
  6  * - Register with email+password
  7  */
  8 function DesktopLoginView() {
  9   var that = this;
 10 
 11   this.e = $('<div></div>').attr('id', 'desktoploginview').appendTo('body').css('display', 'none');
 12 
 13   this.e.load('js/login/template.html', function() {
 14     // When the view is loaded, register handlers
 15     $("#password").keypress(function(event) {
 16       if (event.which == 13) {
 17         $("#login").focus().click();
 18       }
 19     });
 20 
 21     $("#login").click(function() {
 22       that.fireLogin($("#email").val(), $("#password").val());
 23     });
 24 
 25     $("#register").click(function() {
 26       that.fireRegister($("#email").val(), $("#password").val());
 27     });
 28   });
 29 };
 30 DesktopLoginView.prototype = new AbstractLoginView();
