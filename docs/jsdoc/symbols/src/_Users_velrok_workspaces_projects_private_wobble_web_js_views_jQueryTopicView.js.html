  3 // UI Functions
  4 var userCache = {}; // id => {id, name, email, img}
  5 function jQueryTopicView() {  // The UI handler for the single topic
  6   this.editingPostId = null;
  7   this.currentTopic = null;
  8 
  9   this.e = $('<div></div>').addClass('widget').attr('id', 'topic_wrapper').appendTo('#widgets');
 10 
 11   this.jTopicReaders = $('<div></div>').attr('id', 'topic_readers').appendTo(this.e);
 12   this.jTopicActions = $('<div></div>').attr('id', 'topic_actions').appendTo(this.e);
 13   this.$messages = $('<div></div>').attr('id', 'topic_messages').appendTo(this.e);
 14   this.jTopicPosts = $('<div></div>').attr('id', 'topic_posts').appendTo(this.e);
 15 
 16   this._renderTopicActions(false);
 17 
 18   // On a window.resize event wait for the transformations to finish (should be done in 300ms) and recalc height
 19   BUS.on('window.resize', function() {
 20     var t = this;
 21     window.setTimeout(function() {
 22       t.onResize();
 23     }, 350);
 24   }, this);
 25 };

 59 jQueryTopicView.prototype.renderTopic = function(topicDetails) {
 60   $("#topic_posts .loading").detach();
 61 
 62   this.currentTopic = topicDetails;
 63 
 64   this._renderTopicActions($(".editing").length > 0);
 65 
 66   if (topicDetails) {
 67     this.setEnabled(true);
 68 
 69     this.jTopicReaders.empty();
 70     // Only cache the writers
 71     for (var i = 0; i<  topicDetails.writers.length; ++i) {
 72       var user = topicDetails.writers[i];
 73       userCache[user.id] = user;
 74     }
 75     // Cache & render the readers
 76     for (var i = 0; i < topicDetails.readers.length; ++i) {
 77       var user = topicDetails.readers[i];
 78       userCache[user.id] = user; // Cache user object (user later to show the user post images)
 79       this._renderReader(user);
 80     }
 81 
 82     this.renderMessages(topicDetails.id, topicDetails.messages);
 83 
 84     this.onResize();
 85 
 86     this.renderPosts(topicDetails);
 87   } else {
 88     this.setEnabled(false);
 89   }
 90 };

 92 jQueryTopicView.prototype.renderMessages = function(topic_id, messages) {
 93     this.$messages.empty();
 94     messages.forEach($.proxy(function(msgObj) {
 95         var msg = msgObj.message;
 96         var message_id = msgObj.message_id;
 97         var str;
 98 
 99         if (msg.type == 'user_added') {
100             str = msg.user_name + ' was added.';
101         } else if (msg.type == 'user_removed') {
102             str = msg.user_name + ' was removed';
103         } else {
104           console.log('Unknown message type: ' + msg.type);
105         }
106 
107         if (str) {
108             var con = $('<div></div>');
109             con.addClass('message');
110             $('<div></div>').html(str).appendTo(con);
111             $('<button></button').text('Remove').click(function() {
112                 console.log('topic_remove_message ' + topic_id + ', ' + message_id);
113                 API.topic_remove_message(topic_id, message_id, function(err, result) {
114                   if (!err) {
115                     con.remove();
116                     con = null;
117                   }
118                 });
119             }).appendTo(con);
120             con.appendTo(this.$messages);
121         }
122     }, this));
123 };

145 jQueryTopicView.prototype.renderPosts = function(topicDetails) {
146   var $scrollToPost = null;
147   for (var i = 0; i < topicDetails.posts.length; i++) {
148     var $post = this.renderPost(topicDetails, topicDetails.posts[i]);
149     if (!$scrollToPost && topicDetails.posts[i].deleted == 0 && topicDetails.posts[i].unread == 1) {
150       $scrollToPost = $post; 
151     }
152   }
153 
154   if ($scrollToPost && this.editingPostId == null) {
155     $(">.post", $scrollToPost)[0].scrollIntoView(false);
156   }
157 };
158 jQueryTopicView.prototype.renderPost = function(topic, post) {
159   var elementPostId = 'post-' + post.id;
160   var that = this;
161 
162   var jPostWrapper = $("#" + elementPostId);
163   if (jPostWrapper.length == 0) {
164     // Post does not exist yet in the UI => Create it
165     jPostWrapper = $(
166       "<div class='post_wrapper'>" +
167       " <div class='post'>" +
168       "   <div class='users'></div>" +
169       "       <div class='time'></div>" +
170       "   <div class='content'></div>" +
171       "   <div class='buttons'></div>" +
172       " </div>" +
173       " <div class='post_replies empty'></div>" +
174       "</div>").attr('id', elementPostId);
175 
176     if (post.parent) {
177       var parentPostId = '#post-' + post.parent;
178       if (post.intended_post == 1) {
179         var parentPost = $(parentPostId + ">.post_replies");
180         var thread = $('<div></div>').addClass('intended_reply_thread').appendTo(parentPost).append(jPostWrapper);
181         if (parentPost.hasClass('empty')) {
182           parentPost.removeClass('empty');
183         } else {
184           thread.addClass('thread_spacer');
185         }
186       } else {
187         jPostWrapper.insertAfter(parentPostId);
188       }
189     } else { // Root post
190       jPostWrapper.appendTo(this.jTopicPosts);
191     }
192 
193   }
194   jPostWrapper.data('post', post); // Refresh the bound post
195 
196   $(">.post", jPostWrapper).off('click').click(function() {
197       // Add the nice green border to any clicked post
198       $("#topic_wrapper .active").removeClass('active');
199       $(this).addClass('active');
200 
201       that.onPostClicked(post);
202     });
203 
204   if (post.deleted != 1) {
205     // Render children
206 
207     var ePostUsers = $(">.post>.users", jPostWrapper);
208     this._renderPostUsers(post, ePostUsers);
209 
210     var ePostContent = $(">.post>.content", jPostWrapper);
211     if (post.id != this.editingPostId) { // Leave the content untouched, if the user is editing it
212       ePostContent.html(post.content);
213     }
214     if (post.unread == 1) {
215       $("<div></div>").addClass('unread').appendTo($(">.post", jPostWrapper));
216     } else {
217       $('>.post>.unread', jPostWrapper).detach();
218     }
219 
220     var ePostTime = $(">.post>.time", jPostWrapper).empty();
221     ePostTime.text(this._renderTime(post.timestamp));
222 
223     var ePostButtons = $(">.post>.buttons", jPostWrapper).empty();
224     this._addDefaultButtons(ePostButtons, post);
225   } else {
226     $(">.post",jPostWrapper).detach();
227   }
228   return jPostWrapper;
229 };

231 jQueryTopicView.prototype._renderTime = function(timestamp) {
232   if (!timestamp) {
233     return "";
234   }
235   // NOTE: This format the date in the german way (localized): dd.mm.yyyy hh24:mi
236   var createdAt = new Date(timestamp * 1000), now = new Date();
237   var hours = createdAt.getHours();
238   if (hours < 10) {
239     hours = "0" + hours;
240   }
241   var minutes = createdAt.getMinutes();
242   if (minutes < 10) {
243     minutes = "0" + minutes;
244   }
245   var time = hours + ":" + minutes;
246 
247   var month = createdAt.getMonth() + 1;
248   if (month < 0){
249     month = "0" + month;
250   }
251 
252   if (createdAt.getYear() == now.getYear() &&
253      createdAt.getMonth() == now.getMonth() &&
254      createdAt.getDate() == now.getDate()) { // This post is from today, only show the time
255     return time;
256   } else {
257     return createdAt.getDate() + "." + month + "."+ (1900 + createdAt.getYear()) + " " + time;
258   }
259 };

261 jQueryTopicView.prototype._renderPostUsers = function(post, postElement) {
262   if (postElement == null) {
263     postElement = $("#post-" + post.id + ">.post>.users");
264   }
265   postElement.empty();
266 
267   var minHeight = 16;
268   if (post.id != ROOT_ID) { // No user icons for the root
269     var size = post.users.length == 1 ? 25 : 21;
270     for (var i = 0; i < post.users.length; i++) {
271       var postUserId = post.users[i];
272       var template = "<img width='{{size}}' height='{{size}}' src='http://gravatar.com/avatar/{{img}}?s={{size}}' title='{{name}}'/>";
273       postElement.append(Mustache.to_html(template, {
274         'img': userCache[postUserId].img,
275         'name': userCache[postUserId].name,
276         'size': size
277       }));
278     }
279     minHeight = size;
280   }
281 
282   // Part 2: Render the author names
283   function name(index) {
284     if (userCache[post.users[index]].id == API.user_id()) {
285       return "Me";
286     } else {
287       return userCache[post.users[index]].name;
288     }
289   };
290   var apiUserId = API.user_id();
291   var authorLine = null;
292   if (post.users.length == 1 && (post.id != ROOT_ID || post.users[0] != apiUserId) /* no authorline for ourself */) {
293     authorLine = name(0);
294   } else if (post.users.length == 2) {
295     authorLine = name(0) + " and " + name(1);
296   } else if (post.users.length == 3) {
297     authorLine = name(0) + ", " + name(1) + " and " + name(2);
298   } else if (post.users.length >= 4) {
299     authorLine = name(0) + ", " + name(1) + " and " + (post.users.length-2) + " more";
300   } else {
301     // NO authorlines (also no icons) => No min height
302     minHeight = null;
303   }
304   postElement.append($("<span class='names'></span>").text(authorLine));
305   if (minHeight) postElement.css('min-height', minHeight);
306 
307 };

309 jQueryTopicView.prototype.removePost = function(post) {
310   var jpost = $('#post-' + post.id + ">.post");
311   var parent = jpost.parent(); // postwrapper
312   var container = parent.parent(); // #topic_posts or .intended_reply_thread
313   jpost.detach();
314   if ($('>.post_replies', parent).hasClass('empty')) {
315     parent.detach();
316   }
317   if (container.hasClass('intended_reply_thread') && container.children().size() == 0) {
318     container.detach();
319   }

323 jQueryTopicView.prototype.openEditor = function(post) {
324   this.closeEditor(); // Close any open editor there is
325 
326   var that = this;
327 
328   this.editingPostId = post.id;
329   this.onStartPostEdit(post); // Fire notifier event
330 
331   var jpost = $("#post-" + post.id + ">.post");
332   jpost.click(); // Mark active
333 
334   var eContent = $(">.content", jpost).attr('contenteditable', 'true');
335   eContent.addClass('editing')./* makes formatting buttons unusable: blur(submitPostEditing).*/focus();
336   eContent.keydown(function(event) {
337     if (event.shiftKey && event.which == 13) {
338       // Focus the button, otherwise there is a rendering bug in chrome when removing the
339       // contenteditable from the div while it has focus (the editing border does not get removed, until you click somewhere)
340       $(">.buttons>button", jpost).focus();
341       that.closeEditor();
342 
343       event.stopPropagation();
344       event.preventDefault();
345       event.stopImmediatePropagation();
346     }
347   });
348   this._addDefaultButtons($(">.buttons", jpost).empty(), post);
349 
350   this._renderTopicActions(true);
351 };

353 jQueryTopicView.prototype.closeEditor = function() {
354   var jediting = $(".editing");
355   this.editingPostId = null;
356 
357   if (jediting.length > 0) {
358     this._renderTopicActions(false);
359     jediting.attr('contenteditable', 'false').removeClass('editing');
360 
361     var post = jediting.parentsUntil('.post_wrapper').parent().data('post');
362     if (post) {
363       var jpost = $("#post-" + post.id + ">.post");
364       this._addDefaultButtons($(".buttons", jpost).empty(), post);
365 
366       var content = $("#post-" + post.id + " .content").html();
367       this.onStopPostEdit(post, content);
368     }
369   }
370 };

372 jQueryTopicView.prototype._addDefaultButtons = function(jbuttons, post) {
373   var that = this;
374 
375   if (this.editingPostId == post.id) {
376     $("<button><b>Done</b> <span style='font-size:small; color:gray'>[Shift+Enter]</span></button>").appendTo(jbuttons).click(function(event) {
377       that.closeEditor();
378 
379       event.stopPropagation();
380       event.preventDefault();
381       event.stopImmediatePropagation();
382 
383     });
384   } else {
385     jbuttons.append($("<button>Edit</button>").click(function(event) {
386       that.openEditor(post);
387     }));
388     jbuttons.append($("<button>Reply</button>").click(function(event) {
389       event.stopPropagation();
390       event.preventDefault();
391       event.stopImmediatePropagation();
392       // If the post-wrapper has a next-sibling, we create an intended-reply
393       if ($(this).closest('.post_wrapper').next('.post_wrapper').size() > 0) {
394         that.onIntendedReplyPost(post);
395       } else {
396         that.onReplyPost(post);
397       }
398     }));
399     if (post.id != ROOT_ID) { // You cannot delete the root
400       $("<button>Delete</button>").appendTo(jbuttons).click(function() {
401         if (window.confirm('Are you sure to delete this post?')) {
402           that.onDeletePost(post);
403         }
404       });
405     }
406   }
407 
408   if (post.locked) {
409     $("button", jbuttons).attr('disabled', 'disabled');
410   }
411   return jbuttons;
412 };
